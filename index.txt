  <script>
    const persons = document.querySelectorAll('.person');
    const overlay = document.getElementById('overlay');
    const closeBtn = document.getElementById('closeBtn');
    const oName = document.getElementById('overlayName');
    const oRole = document.getElementById('overlayRole');
    const oDesc = document.getElementById('overlayDesc');
    const oImg = document.getElementById('overlayImg');
    const oAudio = document.getElementById('overlayAudio');

    persons.forEach(p => {
      p.addEventListener('click', () => {
        oName.textContent = p.dataset.name;
        oRole.textContent = p.dataset.role;
        oDesc.textContent = p.dataset.desc;
        oImg.src = p.dataset.img || p.querySelector('img').src;
        oAudio.src = p.dataset.audio;
        overlay.style.display = 'flex';
      });
    });

    closeBtn.addEventListener('click', () => {
      overlay.style.display = 'none';
      oAudio.pause();
    });
  </script>
